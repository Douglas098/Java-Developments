package tfl_system;
import java.util.Scanner;
import static tfl_system.TfL_System.main;

public class PinkMarkerClass {

    private String confirm;
    protected boolean Pink;
	
    public PinkMarkerClass(){
		
        String Y = "Y",		N = "N",
                   y = "y",		n = "n";

        Scanner scan = new Scanner(System.in);
		
        System.out.printf("Did you traveled through zone 1?\nY or N: ");
        confirm = scan.next();
        
        if ((confirm.equals(Y) || confirm.equals(y)) || ((confirm.equals(N) || confirm.equals(n)))){
            
            if (confirm.equals(Y) || confirm.equals(y)){
            Pink = true;
            }
           if (confirm.equals(N) || confirm.equals(n)){
            Pink = false;
            }
        } else {
            System.out.println("\nYou entered with an invalid char. Please try again.\n");
            main(new String[]{});
            System.exit(0);
        }
		
        
    }
	
    public boolean PinkMarker(){
        return Pink;
    }
}
