package tfl_system;
import java.util.Calendar;

public class ClockClass {
	
         protected final int On = 1,                            //========== if 1, on peak
                                      Off = 0;                          //=========== if 0, off peak
         
         protected final int sevenHour = 7,                     //============ Unchangeable peak hours
                                      tenHour = 9,
                                      sixteenHour = 16,
                                      twentyHour = 19;
         
         Calendar cal = Calendar.getInstance();                         //========== Calling the callendar, and defining the week, hour and minutes

         Integer atualWeekDay = cal.get(Calendar.DAY_OF_WEEK);
         Integer atualHour = cal.get(Calendar.HOUR_OF_DAY);
         Integer atualMinute = cal.get(Calendar.MINUTE);
	
         public int Peak_Time(){
			
             if ((this.atualHour >= this.sevenHour && this.atualHour <= this.tenHour || this.atualHour >= this.sixteenHour && this.atualHour <= this.twentyHour) && (atualWeekDay != 1 && atualWeekDay != 7)){              //=========== defining the rules of peak time
                 return this.On;
             } else {
                 return this.Off;
             }
         }
}
