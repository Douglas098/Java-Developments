package tfl_system;
import java.util.Scanner;
import static tfl_system.TfL_System.main;

public class EntryExitClass{

         protected int stationEntry, stationExit;               //========= Defining the entry and the exit
    
         public EntryExitClass(){
            try {                                                                                   //========== Using try, to catch...
                  Scanner scan = new Scanner(System.in);
	
                  System.out.printf("\nEntry with the number of entry station: ");
                  stationEntry = scan.nextInt();
        
                   System.out.printf("Entry with the number of exit station: ");
                  stationExit = scan.nextInt();
            }
            catch (java.util.InputMismatchException e) {                                                            //============ ...errors about to insert letters
                  System.out.printf("\nYou entered with a letter. Try again please.\n\n");
                  main(new String[]{});
                  System.exit(0);
            }
            catch (java.lang.StackOverflowError e) {                                                                            //========= ...and stack overflow
                  System.out.printf("\nYou entered with a nonexistent number. Try again please.\n\n");
                  main(new String[]{});
                  System.exit(0);
            }
        }
          
         public int Entry(){                            //=============== Returning the results
             return this.stationEntry;
         }
         public int Exit(){
            return this.stationExit;
    }
}
